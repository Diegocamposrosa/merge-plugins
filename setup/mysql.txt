CREATE TABLE users 
(
ip VARCHAR(30) NOT NULL,
username VARCHAR(30),
auth VARCHAR(32),
firstSeen TIMESTAMP, 
lastSeen TIMESTAMP,
timesSeen INT,
download BIGINT,
upload BIGINT,
timesRun INT,
mergesBuilt INT,
pluginsChecked INT,
pluginsMerged INT,
reportsSubmitted INT
);

CREATE TABLE blacklist
(
ip VARCHAR(30) NOT NULL,
username VARCHAR(30),
created TIMESTAMP,
expires TIMESTAMP
);
